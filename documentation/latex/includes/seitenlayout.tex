% Beautify quotations.
\newcommand{\epigraph}[2]{%
  \begin{center}
  \colorbox{lightdhbwred}{%
    \parbox{.9\textwidth}{%
    	\hrule \vskip 1em {\hugequote} \vskip -.5em
    	\parindent 2.2em
    	#1\begin{flushright}\textsc{#2}\end{flushright}
    	\hrule
    }
  }
  \end{center}
  \bigskip
}
